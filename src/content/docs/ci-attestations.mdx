  return (
    <>
      <p className="text-lg text-gray-200 mb-6">
        API keys support private upload, report polling, draft creation, public
        attestation publishing, and README badge updates from CI. The pattern below matches the
        typical workflow: build your release artifact, upload those exact bytes, publish the attestation,
        and write a compact split-pill badge into the README.
      </p>

      <div className="rounded-lg border border-teal-900/50 bg-teal-950/20 p-4 mb-8">
        <h3 className="text-teal-300 font-semibold mb-2">What this flow does</h3>
        <ul className="list-disc pl-6 space-y-2 text-sm text-gray-300">
          <li>Uploads the exact release artifact to the MLVScan API with your API key.</li>
          <li>Polls the private report until the scan is complete, including the initial submission materialization delay.</li>
          <li>Creates or reuses a draft attestation bound to those exact bytes.</li>
          <li>Publishes the attestation and writes the public URL and badge URL into the workflow summary.</li>
          <li>Updates an existing README attestation badge or inserts one after the current top badge block.</li>
        </ul>
      </div>

      <h2 id="requirements">Requirements</h2>
      <ul className="list-disc pl-6 space-y-2 text-gray-300 mb-6">
        <li>Create an API key in the dashboard under <code>/dashboard/access</code>.</li>
        <li>Store it as a GitHub secret such as <code>MLVSCAN_API_KEY</code>.</li>
        <li>If you use explicit scopes, the key needs <code>uploads:write</code>, <code>reports:read</code>, and <code>attestations:write</code>.</li>
        <li>
          If you use a custom API domain, ensure the base URL your CI calls is reachable without interactive bot checks or browser challenges.
          If needed, set a repository variable like <code>MLVSCAN_API_BASE_URL</code> to a CI-reachable hostname for your API.
        </li>
        <li>Keep using the direct <code>POST /files</code> path only for artifacts under the API upload limit. Larger files should switch to <code>/files/upload_url</code>.</li>
        <li>Authenticated uploads become account-owned. Only the same account can read the private report or turn that submission into an attestation.</li>
      </ul>

      <div className="rounded-lg border border-slate-700 bg-slate-900/60 p-4 mb-8">
        <h3 className="text-white font-semibold mb-2">Auth Header</h3>
        <p className="text-sm text-gray-300 mb-3">
          Every automation request must include your API key in the <code>X-API-Key</code> header.
        </p>
        <pre className="bg-gray-950 rounded-lg p-3 overflow-x-auto"><code>{`curl -H "X-API-Key: $MLVSCAN_API_KEY" \\
  https://api.mlvscan.com/automation/attestations`}</code></pre>
      </div>

      <h2 id="github-actions-template">GitHub Actions Example</h2>
      <pre className="bg-gray-800 rounded-lg p-4 mb-6 overflow-x-auto"><code>{`name: Publish MLVScan attestation

on:
  workflow_dispatch:
  release:
    types:
      - published

jobs:
  attest:
    runs-on: ubuntu-latest
    permissions:
      contents: write
    env:
      MLVSCAN_API_BASE_URL: \${{ vars.MLVSCAN_API_BASE_URL }}
    steps:
      - name: Resolve target ref
        id: ref
        run: |
          if [ "\${{ github.event_name }}" = "release" ]; then
            echo "target_ref=\${{ github.event.release.target_commitish }}" >> "$GITHUB_OUTPUT"
          else
            echo "target_ref=\${{ github.ref_name }}" >> "$GITHUB_OUTPUT"
          fi

      - uses: actions/checkout@v4
        with:
          token: \${{ secrets.GITHUB_TOKEN }}
          fetch-depth: 0
          ref: \${{ steps.ref.outputs.target_ref }}

      - name: Resolve API base URL
        id: api-base
        run: |
          resolved_base_url="\${MLVSCAN_API_BASE_URL:-https://api.mlvscan.com}"
          echo "base_url=$resolved_base_url" >> "$GITHUB_OUTPUT"

      - name: Build release artifact
        run: dotnet build path/to/YourProject.csproj -c Release

      - name: Publish MLVScan attestation
        env:
          MLVSCAN_API_KEY: \${{ secrets.MLVSCAN_API_KEY }}
        run: |
          set -euo pipefail
          artifact_path="path/to/your/artifact.dll"

          if [ ! -f "$artifact_path" ]; then
            echo "Artifact not found: $artifact_path" >&2
            exit 1
          fi

          upload_status=$(curl -sS -o upload.json -w "%{http_code}" \
            -H "X-API-Key: $MLVSCAN_API_KEY" \
            -F "file=@$artifact_path" \
            "\${{ steps.api-base.outputs.base_url }}/files")

          upload_json=$(cat upload.json)
          if [ "$upload_status" -lt 200 ] || [ "$upload_status" -ge 300 ]; then
            echo "$upload_json"
            exit 1
          fi

          submission_id=$(echo "$upload_json" | jq -r '.data.id')
          version="\${{ github.ref_name }}"

          for attempt in $(seq 1 60); do
            report_status=$(curl -sS -o report.json -w "%{http_code}" \
              -H "X-API-Key: $MLVSCAN_API_KEY" \
              "\${{ steps.api-base.outputs.base_url }}/reports/$submission_id")

            report_json=$(cat report.json)

            if [ "$report_status" = "404" ]; then
              error_message=$(echo "$report_json" | jq -r '.error // empty')
              if [ "$error_message" = "Submission not found" ] || [ "$error_message" = "Report not found" ]; then
                sleep 10
                continue
              fi
            fi

            if [ "$report_status" -lt 200 ] || [ "$report_status" -ge 300 ]; then
              echo "$report_json"
              exit 1
            fi

            status=$(echo "$report_json" | jq -r '.status')

            if [ "$status" = "completed" ]; then
              break
            fi

            if [ "$status" = "failed" ]; then
              echo "$report_json" | jq
              exit 1
            fi

            sleep 10
          done

          if [ "\${{ github.event_name }}" = "release" ]; then
            canonical_source_url="\${{ github.event.release.html_url }}"
          else
            canonical_source_url="https://github.com/\${{ github.repository }}/commit/\${{ github.sha }}"
          fi

          draft_payload=$(jq -n \
            --arg submissionId "$submission_id" \
            --arg publicDisplayName "Release artifact $version" \
            --arg canonicalSourceUrl "$canonical_source_url" \
            '{
              submissionId: $submissionId,
              publicDisplayName: $publicDisplayName,
              canonicalSourceUrl: $canonicalSourceUrl
            }')

          draft_json=$(curl -fsS \
            -X POST \
            -H "X-API-Key: $MLVSCAN_API_KEY" \
            -H "Content-Type: application/json" \
            -d "$draft_payload" \
            "\${{ steps.api-base.outputs.base_url }}/automation/attestations")

          attestation_id=$(echo "$draft_json" | jq -r '.id')

          published_json=$(curl -fsS \
            -X POST \
            -H "X-API-Key: $MLVSCAN_API_KEY" \
            "\${{ steps.api-base.outputs.base_url }}/automation/attestations/$attestation_id/publish")

          public_url=$(echo "$published_json" | jq -r '.publicUrl')
          badge_url=$(echo "$published_json" | jq -r '.badgeUrl')
          readme_badge="[![MLVScan Attestation]($badge_url?style=split-pill)]($public_url)"

          if grep -q '^\[!\[MLVScan Attestation\]' README.md; then
            badge_line=$(grep -n '^\[!\[MLVScan Attestation\]' README.md | head -1 | cut -d: -f1)
            awk -v line="$badge_line" -v new_badge="$readme_badge" 'NR==line {print new_badge; next} {print}' README.md > README.md.tmp
            mv README.md.tmp README.md
          else
            insert_after=$(awk '
              NR == 1 { next }
              /^[[:space:]]*$/ {
                if (last_badge > 0) {
                  print last_badge
                  exit
                }
                next
              }
              /^\[!\[/ {
                last_badge = NR
                next
              }
              {
                if (last_badge > 0) {
                  print last_badge
                } else {
                  print 1
                }
                exit
              }
              END {
                if (last_badge > 0) {
                  print last_badge
                } else {
                  print 1
                }
              }
            ' README.md | head -1)

            awk -v line="$insert_after" -v new_badge="$readme_badge" '
              { print }
              NR == line { print new_badge }
            ' README.md > README.md.tmp
            mv README.md.tmp README.md
          fi

          {
            echo "## MLVScan attestation"
            echo
            echo "- Public URL: $public_url"
            echo "- Badge URL: $badge_url"
          } >> "$GITHUB_STEP_SUMMARY"`}</code></pre>
