var fileOption = new Option<string>("--file") { Required = true };
var failOnOption = new Option<string>("--fail-on");

root.Options.Add(fileOption);
root.Options.Add(failOnOption);

root.SetAction(parseResult =>
{
    var file = parseResult.GetValue(fileOption);
    var failOn = parseResult.GetValue(failOnOption);
    return RunScan(file, failOn);
});

return root.Parse(args).Invoke();`}</code></pre>

    await RunScanAsync(file, cancellationToken);
    return 0;
});

return await root.Parse(args).InvokeAsync();`}</code></pre>
